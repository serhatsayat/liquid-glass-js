Usage Examples

Creating Buttons

// Create a primary button const primaryBtn = LiquidGlass.createButton('Click Me', () => { console.log('Button clicked!'); }, { variant: 'primary' }); // Create with factory method const { component: button } = LiquidGlass.create('button', element, { variant: 'secondary', size: 'large', morphing: true }); // Using static methods const floatingBtn = GlassButton.createFloating('Floating Button', handleClick);

Creating Cards

// Create a simple card const card = LiquidGlass.createCard('<h3>Title</h3><p>Content</p>', { variant: 'elevated', clickable: true, onClick: (e, card) => console.log('Card clicked') }); // Card with header and footer const advancedCard = GlassCard.createWithHeader( '<h2>Card Title</h2>', '<p>Card content goes here</p>', '<button>Action</button>' );

Creating Modals

// Create an alert modal const alert = LiquidGlass.createAlertModal( 'Success!', 'Your action was completed successfully.', () => console.log('OK clicked') ); // Create a form modal const form = LiquidGlass.createFormModal( 'Contact Form', [ { name: 'name', label: 'Name', required: true }, { name: 'email', label: 'Email', type: 'email', required: true }, { name: 'message', label: 'Message', type: 'textarea' } ], (data) => console.log('Form submitted:', data) );
